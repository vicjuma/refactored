    }
  </style>
</head>
<body>
  <div id="video-container"></div>

  <script src="https://code.jquery.com/jquery-3.6.0.min.js"></script>
  <script>
    // Sample video data
    const videos = [
      {
        id: 1,
        title: "Video 1",
        url: "https://example.com/video1.mp4"
      },
      {
        id: 2,
        title: "Video 2",
        url: "https://example.com/video2.mp4"
      },
      {
        id: 3,
        title: "Video 3",
        url: "https://example.com/video3.mp4"
      }
    ];

    // Function to generate video thumbnails
    function generateThumbnails() {
      const videoContainer = document.getElementById("video-container");

      videos.forEach(video => {
        const thumbnail = document.createElement("div");
        thumbnail.classList.add("video-thumbnail");
        thumbnail.innerHTML = `<img src="thumbnail_${video.id}.jpg" alt="${video.title}" />`;
        
        // Attach click event listener
        thumbnail.addEventListener("click", () => {
          playVideo(video);
        });

        videoContainer.appendChild(thumbnail);
      });
    }

    // Function to play the selected video
    function playVideo(video) {
      // Clear the video container
      const videoContainer = document.getElementById("video-container");
      videoContainer.innerHTML = "";

      // Create video element and set attributes
      const videoElement = document.createElement("video");
      videoElement.setAttribute("controls", true);
      videoElement.setAttribute("autoplay", true);
      videoElement.setAttribute("src", video.url);

      // Append video element to the container
      videoContainer.appendChild(videoElement);
    }

    // Call the function to generate video thumbnails
    $(document).ready(() => {
      generateThumbnails();
    });
  </script>
</body>
</html>

    In this code, we have an array of sample video data represented by the videos variable. The generateThumbnails function dynamically generates video thumbnails based on the data and attaches a click event listener to each thumbnail. When a thumbnail is clicked, the playVideo function is called, which replaces the video thumbnails with the selected video's player.

    Remember to replace the URLs of the video files and thumbnail images with your actual video and thumbnail URLs. You can customize the CSS styles in the <style> section to match your desired design.
